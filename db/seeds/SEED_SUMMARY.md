-- Check user dengan events yang mereka buat
SELECT u.username, u.name, COUNT(e.id) as events_created
FROM users u
LEFT JOIN events e ON u.id = e.host_id
GROUP BY u.id, u.username, u.name
ORDER BY events_created DESC;

-- Check user dengan posts
SELECT u.username, u.name, COUNT(p.id) as posts_count
FROM users u
LEFT JOIN posts p ON u.id = p.author_id
GROUP BY u.id, u.username, u.name
ORDER BY posts_count DESC;

-- Check followers/following per user
SELECT u.username, u.name,
  (SELECT COUNT(*) FROM follows WHERE following_id = u.id) as followers,
  (SELECT COUNT(*) FROM follows WHERE follower_id = u.id) as following
FROM users u
ORDER BY followers DESC;

-- Check events yang user ikuti
SELECT u.username, u.name, COUNT(ea.id) as events_attended
FROM users u
LEFT JOIN event_attendees ea ON u.id = ea.user_id
WHERE ea.status = 'confirmed'
GROUP BY u.id, u.username, u.name
ORDER BY events_attended DESC;
